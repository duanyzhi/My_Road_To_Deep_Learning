列表extend、append
a = ['1', '2', '3']
b = a[::]
c = a[::]
d = a             # d = a, d = a*2这些操作都会使得a、d绑定
a.append('4')
print(a)
>> ['1', '2', '3', '4']
b.append(['4', '5'])
print(b)
>> ['1', '2', '3', ['4', '5']]
c.extend(['4', '5'])
print(c)
>> ['1', '2', '3', '4', '5']
d.append('6')
print(d)
>> ['1', '2', '3', '4', '6']
print(a)
>> ['1', '2', '3', '4', '6']    # 这种直接等于 a和d操作一样了
